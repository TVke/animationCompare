Web animation

CSS

@keyframes <name>{
  from{left:0;}
  to{left:100%;}
}

.ball{
  animate: <name> 3s linear infinite;
}

JS

var counter = 0, step = (100 - 0) / (3 / (1 / 50));
var animate = setInterval( function(){
  if( 0 + (counter * step) <= 100 ){
    ball.setAttribute("style", "left:" + roundNumber(0 + (counter * step), 2) + "%");
    ++counter;
  }else{
    ball.setAttribute("style", "left:" + 100 + "%";
    clearInterval(animate);
  }
}
